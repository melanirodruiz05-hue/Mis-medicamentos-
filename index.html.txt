<option>Pastilla</option>
<option>Polvo</option>
</select>

<input type="number" id="dosis" placeholder="Veces al día">
<input type="number" id="cantidad" placeholder="Cantidad total">

<button onclick="agregar()">Guardar 💖</button>

<div id="lista"></div>

<h2>📊 Historial</h2>
<div id="historial"></div>

<script>
let meds = JSON.parse(localStorage.getItem("meds")) || [];
let historial = JSON.parse(localStorage.getItem("historial")) || [];

function guardar(){
    localStorage.setItem("meds", JSON.stringify(meds));
    localStorage.setItem("historial", JSON.stringify(historial));
}

function agregar(){
    meds.push({
        nombre: nombre.value,
        inicio: inicio.value,
        fin: fin.value,
        tipo: tipo.value,
        dosis: dosis.value,
        cantidad: cantidad.value
    });
    guardar();
    render();
}

function tomar(i){
    meds[i].cantidad--;

    historial.push({
        nombre: meds[i].nombre,
        fecha: new Date().toLocaleString()
    });

    if(meds[i].cantidad == 5){
        alert("⚠️ Solo te quedan 5 de " + meds[i].nombre);
    }

    guardar();
    render();
}

function render(){
    lista.innerHTML="";
    historial.innerHTML="";

    meds.forEach((m,i)=>{
        lista.innerHTML+=`
        <div class="card">
        <b>${m.nombre}</b><br>
        ${m.tipo} | ${m.dosis} al día<br>
        Restantes: ${m.cantidad}
        ${m.cantidad <=5 ? '<div class="alerta">⚠️ Pocas unidades</div>' : ''}
        <br>
        <button onclick="tomar(${i})">Tomé una</button>
        </div>`;
    });

    historial.forEach(h=>{
        historial.innerHTML+=`<div>${h.nombre} - ${h.fecha}</div>`;
    });
}

render();
</script>
